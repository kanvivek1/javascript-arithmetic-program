// Simple Arithmetic Operations using Switch Case function calculator(num1, num2, operator) { let result; switch (operator) { case '+': result = num1 + num2; break; case '-': result = num1 - num2; break; case '*': result = num1 * num2; break; case '/': if (num2 !== 0) { result = num1 / num2; } else { result = "Error! Division by zero."; } break; case '%': result = num1 % num2; break; default: result = "Invalid operator!"; } return result; } // Example usage: console.log(calculator(10, 5, '+')); // Output: 15 console.log(calculator(10, 5, '-')); // Output: 5 console.log(calculator(10, 5, '*')); // Output: 50 console.log(calculator(10, 5, '/')); // Output: 2 console.log(calculator(10, 5, '%')); // Output: 0 console.log(calculator(10, 0, '/')); // Output: Error! Division by zero.
